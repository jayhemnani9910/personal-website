import pandas as pd
import plotly.graph_objects as go

def calculate_xg(shot_event):
    """Calculate expected goals based on shot context."""
    distance = shot_event['distance_to_goal']
    angle = shot_event['shot_angle']
    is_header = shot_event['body_part'] == 'head'

    # Simplified xG model (production would use trained ML model)
    base_xg = 1 / (1 + distance / 10)
    angle_factor = angle / 90  # normalized angle
    header_penalty = 0.7 if is_header else 1.0

    return base_xg * angle_factor * header_penalty

def update_xg_chart(new_event):
    """Incrementally update xG timeline without reprocessing all events."""
    if 'xg_timeline' not in st.session_state:
        st.session_state.xg_timeline = {'home': [], 'away': []}

    if new_event['type'] == 'shot':
        team = new_event['team']
        xg_value = calculate_xg(new_event)

        # Append to cumulative timeline
        st.session_state.xg_timeline[team].append({
            'minute': new_event['minute'],
            'cumulative_xg': sum([e['xg'] for e in st.session_state.xg_timeline[team]]) + xg_value,
            'xg': xg_value
        })

    # Render updated chart
    fig = go.Figure()
    for team in ['home', 'away']:
        data = st.session_state.xg_timeline[team]
        if data:
            df = pd.DataFrame(data)
            fig.add_trace(go.Scatter(
                x=df['minute'],
                y=df['cumulative_xg'],
                mode='lines+markers',
                name=team.upper()
            ))

    fig.update_layout(
        title="Expected Goals Over Time",
        xaxis_title="Match Minute",
        yaxis_title="Cumulative xG"
    )

    return fig

import plotly.graph_objects as go
from scipy.stats import gaussian_kde

def generate_heatmap(player_positions, pitch_dimensions=(105, 68)):
    """Create density heat map from player coordinate stream."""
    if len(player_positions) < 10:
        return None  # Not enough data for meaningful heat map

    # Extract x, y coordinates
    x = np.array([pos['x'] for pos in player_positions])
    y = np.array([pos['y'] for pos in player_positions])

    # Create KDE for smooth density estimation
    positions = np.vstack([x, y])
    kernel = gaussian_kde(positions)

    # Generate grid for heat map
    x_grid = np.linspace(0, pitch_dimensions[0], 50)
    y_grid = np.linspace(0, pitch_dimensions[1], 50)
    X, Y = np.meshgrid(x_grid, y_grid)

    # Evaluate density at grid points
    grid_coords = np.vstack([X.ravel(), Y.ravel()])
    Z = kernel(grid_coords).reshape(X.shape)

    # Create Plotly heatmap overlay
    fig = go.Figure(data=go.Heatmap(
        x=x_grid,
        y=y_grid,
        z=Z,
        colorscale='Reds',
        opacity=0.6,
        showscale=False
    ))

    # Add pitch markings
    fig.update_layout(
        xaxis=dict(range=[0, pitch_dimensions[0]], title="Field Length (m)"),
        yaxis=dict(range=[0, pitch_dimensions[1]], title="Field Width (m)"),
        title="Player Pressure Heat Map"
    )

    return fig`
