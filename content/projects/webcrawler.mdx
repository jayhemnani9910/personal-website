                print('Starting crawl cycle')
                sw.run_cycle()  # Iterates all sites, archives pages

    - title: "Content Hash Function (utils.py)"
      language: "python"

        import hashlib

        def hash_text(text: str) -> str:
            """Compute SHA-256 hash of text content."""
            h = hashlib.sha256()
            if isinstance(text, str):
                text = text.encode('utf-8')
            h.update(text)
            return h.hexdigest()
      explanation: "Simple SHA-256 hashing utility. Returns hex digest for easy storage and comparison. The key is that 'text' is pre-processed by readability-lxml to extract main content and strip dynamic elements, ensuring stable hashes across crawls."
